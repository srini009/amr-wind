71 template <typename ActTrait>
72 struct ComputeForceOp<
73  ActTrait,
74  ActSrcLine,
75  typename std::enable_if<
76  std::is_base_of<WingType, ActTrait>::value &&
77  UseDefaultOp<ActTrait, ActSrcLine>::compute_force>::type>
78 {
79  void operator()(typename ActTrait::DataType& data)
80  {
81  auto& grid = data.grid();
82  auto& wdata = data.meta();
83  const int npts = wdata.num_pts;
84  const auto& dx = wdata.dx;
85  const auto& chord = wdata.chord;
86  const auto& aflookup = airfoil_lookup<ActTrait>(data);
87 
88  amrex::Real total_lift = 0.0;
89  amrex::Real total_drag = 0.0;
90  for (int ip = 0; ip < npts; ++ip) {
91  const auto& tmat = grid.orientation[ip];
92  // Effective velocity at the wing control point in local frame
93  auto wvel = tmat & grid.vel[ip];
94  // Set spanwise component to zero to get a pure 2D velocity
95  wvel.y() = 0.0;
96 
97  const auto vmag = vs::mag(wvel);
98  const auto aoa = std::atan2(wvel.z(), wvel.x());
99 
100  // Make up some Cl, Cd values
101  amrex::Real cl, cd;
102  aflookup(aoa, cl, cd);
103 
104  // Assume unit chord
105  const auto qval = 0.5 * vmag * vmag * chord[ip] * dx[ip];
106  const auto lift = qval * cl;
107  const auto drag = qval * cd;
108  // Determine unit vector parallel and perpendicular to velocity
109  // vector
110  const auto drag_dir = wvel.unit() & tmat;
111  const auto lift_dir = drag_dir ^ tmat.y();
112 
113  // Compute force on fluid from this section of wing
114  grid.force[ip] = -(lift_dir * lift + drag * drag_dir);
115 
116  // Assign values for output
117  wdata.vel_rel[ip] = wvel;
118  wdata.aoa[ip] = amr_wind::utils::degrees(aoa);
119  wdata.cl[ip] = cl;
120  wdata.cd[ip] = cd;
121 
122  total_lift += lift;
123  total_drag += drag;
124  }
125 
126  wdata.lift = total_lift;
127  wdata.drag = total_drag;
128  }
129 };
