105 struct UpdateVelOp<UniformCt, ActSrcDisk>
106 {
107  void operator()(UniformCt::DataType& data)
108  {
109  auto& meta = data.meta();
110  const auto& grid = data.grid();
111  meta.reference_velocity = {0.0, 0.0, 0.0};

116  for (int i = 0; i < np; i++) {
117  refVel = refVel + grid.vel[i];
118  diskVel = diskVel + grid.vel[i + np];
119  }
120  refVel /= np;

125 // compute the total force over a given radial section
126 // this will then be spread uniformly over that area in the
127 // projection step
128 template <>
129 struct ComputeForceOp<UniformCt, ActSrcDisk>
130 {
131  void operator()(UniformCt::DataType& data)
132  {
133  auto& grid = data.grid();
134  auto& ddata = data.meta();
135  const auto& normal = ddata.normal_vec;
136  const amrex::Real rho = ddata.density;
137  // only use velocity components normal to disk
138  // make sure uInfSqr is always positive by squaring the projection of
139  // reference_velocity onto the normal
140  const amrex::Real uInfSqr = (ddata.reference_velocity & normal) *
141  (ddata.reference_velocity & normal);
142 
143  ddata.current_ct = ::amr_wind::interp::linear(
144  ddata.table_velocity, ddata.thrust_coeff,
145  std::sqrt(std::abs(uInfSqr)));
146 
147  const int npts = ddata.num_force_pts;
148  const amrex::Real aeroPressure = 0.5 * uInfSqr * rho * ddata.current_ct;
149  const amrex::Real dr = ddata.dr;
150 
151  for (int ip = 0; ip < npts; ++ip) {
152  const amrex::Real r = (ip + 0.5) * dr;
153  const amrex::Real rp = r + dr * 0.5;
154  const amrex::Real rm = r - dr * 0.5;
155  const amrex::Real a = ::amr_wind::utils::pi() * (rp * rp - rm * rm);
156  // disk force should always point in direction of the normal
157  grid.force[ip] = (aeroPressure * a) * normal;
158  }
159  }
160 };
