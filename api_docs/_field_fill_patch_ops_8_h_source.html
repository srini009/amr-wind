182  {
183  const int nstates = amrex::min(m_field.num_time_states(), 2);
184  amrex::Vector<amrex::MultiFab*> ret;
185 
186  // The states in the FieldInfo data are ordered from newest to oldest,

301  void set_inflow(
302  int lev,
303  amrex::Real time,
304  amrex::MultiFab& mfab,
305  const amrex::IntVect& nghost,
306  const FieldState) override
307  {
308  const int ng = nghost[0];
309  const auto& bctype = m_field.bc_type();
310  const auto& geom = m_mesh.Geom(lev);
311  const auto& gdata = geom.data();
312  const auto& domain = geom.growPeriodicDomain(ng);
313  const auto& bcfunc = bc_functor();
314  const auto& ncomp = m_field.num_comp();
315 
316  for (amrex::OrientationIter oit; oit; ++oit) {
317  auto ori = oit();
318  if (bctype[ori] != BC::mass_inflow) continue;
319 
320  const int idir = ori.coordDir();
321  const auto& dbx =
322  ori.isLow() ? amrex::adjCellLo(domain, idir, nghost[idir])
323  : amrex::adjCellHi(domain, idir, nghost[idir]);
324 
325  for (amrex::MFIter mfi(mfab); mfi.isValid(); ++mfi) {
326  const auto& gbx = amrex::grow(mfi.validbox(), nghost);
327  const auto& bx = gbx & dbx;
328  if (!bx.ok()) continue;
329 
330  const auto& marr = mfab[mfi].array();
331  amrex::ParallelFor(
332  bx, [=] AMREX_GPU_DEVICE(
333  const int i, const int j, const int k) noexcept {
334  for (int n = 0; n < ncomp; ++n)
335  bcfunc.set_inflow(
336  {i, j, k}, marr, gdata, time, ori, n);
337  });
338  }
339  }
340  }
