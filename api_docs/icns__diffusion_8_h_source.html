177  void linsys_solve(const amrex::Real dt)
178  {
179  const FieldState fstate = FieldState::New;
180  auto& repo = m_pdefields.repo;
181  auto& field = m_pdefields.field;
182  const auto& density = m_density.state(fstate);
183  const int nlevels = repo.num_active_levels();
184  const int ndim = field.num_comp();
185  auto rhs_ptr = repo.create_scratch_field("rhs", field.num_comp(), 0);
186  const auto& viscosity = m_pdefields.mueff;
187  const auto& geom = repo.mesh().Geom();
188  const amrex::Real alpha = 1.0;
189  const amrex::Real beta = dt;
190 
191  m_solver_scalar->setScalars(alpha, beta);
192  for (int lev = 0; lev < nlevels; ++lev) {
193  m_solver_scalar->setLevelBC(lev, &m_pdefields.field(lev));
194  m_solver_scalar->setACoeffs(lev, density(lev));
195  auto b = diffusion::average_velocity_eta_to_faces(
196  geom[lev], viscosity(lev));
197  m_solver_scalar->setBCoeffs(lev, amrex::GetArrOfConstPtrs(b));
198  }
199 
200  // Always multiply with rho since there is no diffusion term for density
201  for (int lev = 0; lev < nlevels; ++lev) {
202  auto& rhs = (*rhs_ptr)(lev);
203 
204 #ifdef _OPENMP
205 #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
206 #endif
207  for (amrex::MFIter mfi(rhs, amrex::TilingIfNotGPU()); mfi.isValid();
208  ++mfi) {
209  const auto& bx = mfi.tilebox();
210  const auto& rhs_a = rhs.array(mfi);
211  const auto& fld = field(lev).const_array(mfi);
212  const auto& rho = density(lev).const_array(mfi);
213 
214  amrex::ParallelFor(
215  bx, ndim,
216  [=] AMREX_GPU_DEVICE(int i, int j, int k, int n) noexcept {
217  rhs_a(i, j, k, n) = rho(i, j, k) * fld(i, j, k, n);
218  });
219  }
220  }
221 
222  amrex::MLMG mlmg(*m_solver_scalar);
223  m_options(mlmg);
224  mlmg.solve(
225  m_pdefields.field.vec_ptrs(), rhs_ptr->vec_const_ptrs(),
226  m_options.rel_tol, m_options.abs_tol);
227 
228  io::print_mlmg_info(field.name() + "_multicomponent_solve", mlmg);
229  }
