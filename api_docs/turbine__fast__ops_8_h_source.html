231  void init_epsilon(typename TurbineFast::DataType& data)
232  {
233  auto& tdata = data.meta();
234 
235  // Swap order of epsilon based on FAST turbine orientation
236  swap_epsilon(tdata.eps_inp);
237  swap_epsilon(tdata.eps_min);
238  swap_epsilon(tdata.eps_chord);
239  swap_epsilon(tdata.eps_tower);
240 
241  {
242  const auto& cd = tdata.nacelle_cd;
243  const auto& area = tdata.nacelle_area;
244  const auto eps =
245  std::sqrt(2.0 / ::amr_wind::utils::pi() * cd * area);
246 
247  auto& nac_eps = data.grid().epsilon[0];
248  nac_eps.x() = amrex::max(eps, tdata.eps_min.x());
249  nac_eps.y() = amrex::max(eps, tdata.eps_min.y());
250  nac_eps.z() = amrex::max(eps, tdata.eps_min.z());
251  }
252 
253  for (int ib = 0; ib < tdata.num_blades; ++ib) {
254  auto& cv = tdata.blades[ib];
255 
256  for (int i = 0; i < tdata.num_pts_blade; ++i) {
257  const auto eps_crd = tdata.eps_chord * cv.chord[i];
258 
259  for (int n = 0; n < AMREX_SPACEDIM; ++n) {
260  cv.epsilon[i][n] = amrex::max(
261  tdata.eps_min[n], tdata.eps_inp[n], eps_crd[n]);
262  }
263  }
264  }
265  {
266  auto& cv = tdata.tower;
267  for (int i = 0; i < tdata.num_pts_tower; ++i) {
268  for (int n = 0; n < AMREX_SPACEDIM; ++n) {
269  cv.epsilon[i][n] = amrex::max(
270  tdata.eps_min[n], tdata.eps_inp[n], tdata.eps_tower[n]);
271  }
272  }
273  }

284  void check_fast_sim_time(typename TurbineFast::DataType& data)
285  {
286  const auto& time = data.sim().time();
287 
288  // Set OpenFAST end time to be at least as long as the CFD time. User
289  // can choose a longer duration in input file.
290  const amrex::Real stop1 = time.stop_time() > 0.0
291  ? time.stop_time()
292  : std::numeric_limits<amrex::Real>::max();
293  const amrex::Real stop2 = time.stop_time_index() > -1
294  ? time.stop_time_index() * time.deltaT()
295  : std::numeric_limits<amrex::Real>::max();
296  const amrex::Real cfd_stop = amrex::min(stop1, stop2);
297  const amrex::Real cfd_start = time.current_time();
298  const amrex::Real cfd_sim = cfd_stop - cfd_start - 1.0e-6;
299 
300  // Ensure that the user specified stop_time is not shorter than CFD sim
301  const auto& tf = data.meta().fast_data;
302  const amrex::Real fast_sim = tf.stop_time - tf.start_time;
303  AMREX_ALWAYS_ASSERT_WITH_MESSAGE(
304  fast_sim > cfd_sim,
305  "OpenFAST simulation time is shorter than AMR-Wind duration");
306  }

394  void compute_nacelle_force(typename TurbineFast::DataType& data)
395  {
396  if (!data.info().is_root_proc) return;
397 
398  const auto& cd = data.meta().nacelle_cd;
399  const auto& area = data.meta().nacelle_area;
400  const auto& cd_area = cd * area;
401  const auto& fcfd = data.meta().fast_data.from_cfd;
402  const auto& tcfd = data.meta().fast_data.to_cfd;
403  const auto& rho = data.meta().density;
404 
405  const auto& eps = data.grid().epsilon[0].x();
406  vs::Vector vel{fcfd.u[0], fcfd.v[0], fcfd.w[0]};
407  amrex::Real correction = 0.0;
408  if (eps > 0.0) {
409  amrex::Real fac =
410  1.0 -
411  (cd_area) / (2.0 * ::amr_wind::utils::two_pi() * eps * eps);
412  correction = 1.0 / fac;
413  }
414  amrex::Real coeff =
415  0.5 * rho * cd_area * vs::mag(vel) * correction * correction;
416 
417  tcfd.fx[0] = static_cast<float>(coeff * fcfd.u[0]);
418  tcfd.fy[0] = static_cast<float>(coeff * fcfd.v[0]);
419  tcfd.fz[0] = static_cast<float>(coeff * fcfd.w[0]);
420  }

422  void scatter_data(typename TurbineFast::DataType& data)
423  {
424  if (!data.info().actuator_in_proc) return;
425 
426  // Create an MPI transfer buffer that packs all data in one contiguous
427  // array. 3 floats for the position vector, 3 floats for the force
428  // vector, and 9 floats for the orientation matrix = 15 floats per
429  // actuator node.
430  const int dsize = data.grid().pos.size() * 15;
431  amrex::Vector<float> buf(dsize);
432 
433  // Copy data into MPI send/recv buffer from the OpenFAST data structure.
434  // Note, other procs do not have a valid data in those pointers.
435  if (data.info().is_root_proc) {
436  BL_PROFILE(
437  "amr-wind::actuator::ComputeForceOp<TurbineFast>::scatter1");
438  const auto& tocfd = data.meta().fast_data.to_cfd;
439  auto it = buf.begin();
440  std::copy(tocfd.fx, tocfd.fx + tocfd.fx_Len, it);
441  std::advance(it, tocfd.fx_Len);
442  std::copy(tocfd.fy, tocfd.fy + tocfd.fy_Len, it);
443  std::advance(it, tocfd.fy_Len);
444  std::copy(tocfd.fz, tocfd.fz + tocfd.fz_Len, it);
445  std::advance(it, tocfd.fz_Len);
446 
447  std::copy(tocfd.pxForce, tocfd.pxForce + tocfd.pxForce_Len, it);
448  std::advance(it, tocfd.pxForce_Len);
449  std::copy(tocfd.pyForce, tocfd.pyForce + tocfd.pyForce_Len, it);
450  std::advance(it, tocfd.pyForce_Len);
451  std::copy(tocfd.pzForce, tocfd.pzForce + tocfd.pzForce_Len, it);
452  std::advance(it, tocfd.pzForce_Len);
453 
454  // clang-format off
455  std::copy(tocfd.pOrientation,
456  tocfd.pOrientation + tocfd.pOrientation_Len, it);
457  // clang-format on
458  }
459 
460  // Broadcast data to all influenced procs from the root process
461  const auto& procs = data.info().procs;
462  const int tag = 1001;
463  if (data.info().is_root_proc) {
464  BL_PROFILE(
465  "amr-wind::actuator::ComputeForceOp<TurbineFast>::scatter2");
466  for (const int ip : procs) {
467  if (ip == data.info().root_proc) continue;
468 
469  amrex::ParallelDescriptor::Send(
470  buf.data(), dsize, ip, tag, data.meta().tcomm);
471  }
472  } else {
473  BL_PROFILE(
474  "amr-wind::actuator::ComputeForceOp<TurbineFast>::scatter2");
475  amrex::ParallelDescriptor::Recv(
476  buf.data(), dsize, data.info().root_proc, tag,
477  data.meta().tcomm);
478  }
479 
480  // Populate the actuator grid data structures with data from the MPI
481  // send/recv buffer.
482  {
483  BL_PROFILE(
484  "amr-wind::actuator::ComputeForceOp<TurbineFast>::scatter3");
485  const auto& bp = data.info().base_pos;
486  auto& grid = data.grid();
487  const auto& npts = grid.pos.size();
488  const auto& rho = data.meta().density;
489  const size_t ifx = 0;
490  const size_t ify = ifx + npts;
491  const size_t ifz = ify + npts;
492  const size_t ipx = ifz + npts;
493  const size_t ipy = ipx + npts;
494  const size_t ipz = ipy + npts;
495  const size_t iori = ipz + npts;
496 
497  for (int i = 0; i < npts; ++i) {
498  // Aerodynamic force vectors. Flip sign to get force on fluid.
499  // Divide by density as the source term computation will
500  // multiply by density before adding to momentum equation.
501  //
502  grid.force[i].x() =
503  -static_cast<amrex::Real>(buf[ifx + i]) / rho;
504  grid.force[i].y() =
505  -static_cast<amrex::Real>(buf[ify + i]) / rho;
506  grid.force[i].z() =
507  -static_cast<amrex::Real>(buf[ifz + i]) / rho;
508 
509  // Position vectors of the actuator nodes. Add shift to base
510  // locations.
511  grid.pos[i].x() =
512  static_cast<amrex::Real>(buf[ipx + i]) + bp.x();
513  grid.pos[i].y() =
514  static_cast<amrex::Real>(buf[ipy + i]) + bp.y();
515  grid.pos[i].z() =
516  static_cast<amrex::Real>(buf[ipz + i]) + bp.z();
517 
518  // Copy over the orientation matrix
519  //
520  // Note that we transpose the orientation matrix when copying
521  // from OpenFAST to AMR-Wind Tensor data structure. This is done
522  // so that post-multiplication of vector transforms from global
523  // to local reference frame.
524  const size_t off = iori + i * AMREX_SPACEDIM * AMREX_SPACEDIM;
525  for (int j = 0; j < AMREX_SPACEDIM; ++j)
526  for (int k = 0; k < AMREX_SPACEDIM; ++k)
527  grid.orientation[i][j * AMREX_SPACEDIM + k] =
528  static_cast<amrex::Real>(
529  buf[off + j + k * AMREX_SPACEDIM]);
530  }
531 
532  // Extract the rotor center of rotation
533  auto& meta = data.meta();
534  meta.rot_center = grid.pos[0];
535 
536  // Rotor non-rotating reference frame
537  const auto xvec = grid.orientation[0].x().unit();
538  const auto yvec = vs::Vector::khat() ^ xvec;
539  const auto zvec = xvec ^ yvec;
540  meta.rotor_frame.rows(xvec, yvec.unit(), zvec.unit());
541  }
542  }
