92

100  levelset(lev).array(mfi);
101  const amrex::Real eps =
102  std::cbrt(2. * dx[0] * dx[1] * dx[2]);
103  const amrex::Real mu1 = m_mu1;
104  const amrex::Real mu2 = m_mu2;
105  amrex::ParallelFor(
106  vbx,
107  [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
108  amrex::Real smooth_heaviside;
109  if (phi(i, j, k) > eps) {
110  smooth_heaviside = 1.0;
111  } else if (phi(i, j, k) < -eps) {
112  smooth_heaviside = 0.;
113  } else {
114  smooth_heaviside =
115  0.5 *
116  (1.0 + phi(i, j, k) / eps +
117  1.0 / M_PI *
118  std::sin(phi(i, j, k) * M_PI / eps));
119  }
120  visc(i, j, k) = mu1 * smooth_heaviside +
121  mu2 * (1.0 - smooth_heaviside);
122  });
